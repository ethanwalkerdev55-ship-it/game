public partial class cnMissionManager

	private bool bForceCompleteChain;
	private int m_iForceCompleteChainDepth;
	private int m_iForceCompletePendingTaskId;
	private int m_iForceCompleteRetryCount;

	private cnMissionNode GetForceCompleteTarget()
	{
		if (this.SelectMissionTask != null && this.SelectMissionTask.GetMe() != null && this.IsExistTaskInActiveMission(this.SelectMissionTask.GetMe().m_iHTaskID))
		{
			return this.SelectMissionTask;
		}
		cnMissionNode selectedActiveMission = this.GetSelectedActiveMission();
		if (selectedActiveMission != null && selectedActiveMission.GetMe() != null)
		{
			return selectedActiveMission;
		}
		for (int i = 0; i < this.m_ActivateMissionList.Count; i++)
		{
			cnMissionNode cnMissionNode = (cnMissionNode)this.m_ActivateMissionList[i];
			if (cnMissionNode != null && cnMissionNode.GetMe() != null)
			{
				return cnMissionNode;
			}
		}
		return null;
	}

	private void PrepareTaskForForceComplete(cnMissionNode task)
	{
		if (task == null || task.GetMe() == null)
		{
			return;
		}
		for (int i = 0; i < 3; i++)
		{
			task.m_aiCurrentRemainEnemyNum[i] = 0;
			task.m_aiCurrentRemainItemNum[i] = 0;
		}
		MissionElement me = task.GetMe();
		if (me.m_iSTGrantTimer > 0 || me.m_iCSUCheckTimer > 0)
		{
			task.m_fRemainTime = 0f;
			task.m_fLifeTime = 0f;
		}
	}

	private bool NeedsForceCompleteTaskStart(MissionElement me, cnMissionNode task)
	{
		if (me == null || task == null)
		{
			return false;
		}
		if (0 < me.m_iSTGrantTimer || 0 < me.m_iCSUCheckTimer)
		{
			return !this.IsExistTaskInActiveMission(me.m_iHTaskID) || task.GetMissionState() != 1;
		}
		if (0 < me.m_iRequireInstanceID)
		{
			if (!this.IsExistTaskInActiveMission(me.m_iHTaskID))
			{
				if (this.ownstatus != null && this.ownstatus.iInsMapNum != me.m_iRequireInstanceID)
				{
					return false;
				}
				return true;
			}
		}
		return false;
	}

	private int ResolveForceCompleteTerminatorNpcId(MissionElement me)
	{
		if (me == null || me.m_iHTerminatorNPCID == 0)
		{
			return 0;
		}
		if (this.pNpcContainer.SearchTableID(me.m_iHTerminatorNPCID))
		{
			return me.m_iHTerminatorNPCID;
		}
		if (this.myuser != null)
		{
			this.m_NearList.Clear();
			this.GetNearList(this.myuser);
			if (this.GetNearbyNPC(me.m_iHTerminatorNPCID) != null)
			{
				return me.m_iHTerminatorNPCID;
			}
		}
		return 0;
	}

	private void ClearForceCompleteChain()
	{
		this.bForceCompleteChain = false;
		this.m_iForceCompleteChainDepth = 0;
		this.m_iForceCompletePendingTaskId = 0;
		this.m_iForceCompleteRetryCount = 0;
	}

	private bool IsForceCompleteOutsideInstanceBlocker(MissionElement me)
	{
		if (me != null && me.m_iRequireInstanceID > 0)
		{
			if (this.ownstatus != null)
			{
				return this.ownstatus.iInsMapNum != me.m_iRequireInstanceID;
			}
			return true;
		}
		return false;
	}

	private void RequestForceCompleteTaskEnd(cnMissionNode task)
	{
		MissionElement me = null;
		int taskId = 0;
		if (task != null && task.GetMe() != null)
		{
			me = task.GetMe();
			taskId = me.m_iHTaskID;
		}
		else if (0 < this.m_iForceCompletePendingTaskId)
		{
			taskId = this.m_iForceCompletePendingTaskId;
			cnMissionNode pending = this.GetTask(taskId);
			if (pending != null)
			{
				me = pending.GetMe();
			}
		}
		if (taskId <= 0)
		{
			Logger.Log("ForceCompleteV2: abort end packet no task id");
			return;
		}
		int num = this.m_iForceCompleteRetryCount / 3;
		if (num <= 0)
		{
			Logger.Log("ForceCompleteV2: complete without npc gate task " + taskId.ToString());
		}
		else if (num == 1)
		{
			Logger.Log("ForceCompleteV2: retry without npc task " + taskId.ToString());
		}
		else
		{
			Logger.Log("ForceCompleteV2: retry with error flag task " + taskId.ToString());
		}
		bool sendError = (me != null && 0 < me.m_iRequireInstanceID) || 1 < num;
		if (me != null && 0 < me.m_iRequireInstanceID)
		{
			Logger.Log("ForceCompleteV2: instance zone complete task " + taskId.ToString());
		}
		this.ForceCompleteEmitTaskEndPacket(taskId, 0, sendError);
	}

	private void ForceCompleteEmitTaskEndPacket(int taskId, int npcId, bool sendError)
	{
		Logger.Log("ForceCompleteV2: emit enter task " + taskId.ToString() + " npc " + npcId.ToString() + " err " + sendError.ToString());
		sP_CL2FE_REQ_PC_TASK_END pkt = default(sP_CL2FE_REQ_PC_TASK_END);
		pkt.iTaskNum = taskId;
		if (sendError)
		{
			pkt.iEscortNPC_ID = -1;
		}
		if (npcId != 0)
		{
			Transform transform = this.pNpcContainer.SearchTableID(npcId);
			if (transform)
			{
				Status status = transform.GetComponent(typeof(Status)) as Status;
				if (null != status)
				{
					pkt.iNPC_ID = status.iID;
				}
			}
		}
		Logger.Log("ForceCompleteV2: sent complete packet task " + taskId.ToString() + " npc " + npcId.ToString() + " err " + sendError.ToString());
		cnEvent.SendPacket(Marshal.SizeOf(typeof(sP_CL2FE_REQ_PC_TASK_END)), 318767116, pkt);
		this.AddMissionTaskChecker(taskId);
	}

	private void TryForceCompleteChainTask(cnMissionNode task)
	{
		if (task == null || task.GetMe() == null)
		{
			Logger.Log("ForceCompleteV2: abort chain null task");
			this.ClearForceCompleteChain();
			return;
		}
		if (this.m_iForceCompleteChainDepth >= 16)
		{
			Logger.Log("ForceCompleteV2: abort chain depth limit");
			this.ClearForceCompleteChain();
			return;
		}
		MissionElement me = task.GetMe();
		int taskId = me.m_iHTaskID;
		this.PrepareTaskForForceComplete(task);
		bool needsStart = this.NeedsForceCompleteTaskStart(me, task);
		if (0 < me.m_iSTGrantTimer || 0 < me.m_iCSUCheckTimer)
		{
			needsStart = false;
		}
		if (needsStart)
		{
			Logger.Log("ForceCompleteV2: defer start task " + taskId.ToString());
			this.m_iForceCompletePendingTaskId = taskId;
			this.m_iForceCompleteRetryCount = 0;
			this.RequestTaskStart(taskId, 0);
			return;
		}
		if (0 < me.m_iSTGrantTimer || 0 < me.m_iCSUCheckTimer)
		{
			task.SetRemainTime(0f);
			task.m_fLifeTime = 0f;
		}
		this.m_iForceCompleteChainDepth++;
		this.m_iForceCompleteRetryCount = 0;
		bool sendError = 0 < me.m_iRequireInstanceID;
		Logger.Log("ForceCompleteV2: chain emit task " + taskId.ToString() + " err " + sendError.ToString());
		this.ForceCompleteEmitTaskEndPacket(taskId, 0, sendError);
	}

	private bool ForceCompleteOnEndFail(cnMissionNode task, int forceErrorCode)
	{
		if (!this.bForceCompleteChain || task == null || task.GetMe() == null)
		{
			return false;
		}
		MissionElement me = task.GetMe();
		if (forceErrorCode == 13)
		{
			Logger.Log("ForceCompleteV2: abort fatal error 13 on task " + me.m_iHTaskID.ToString());
			this.ClearForceCompleteChain();
			return true;
		}
		if (forceErrorCode == 1 || forceErrorCode == 11 || forceErrorCode == 12)
		{
			this.m_iForceCompleteRetryCount++;
			if (this.m_iForceCompleteRetryCount > 9)
			{
				Logger.Log("ForceCompleteV2: max retries exceeded on task " + me.m_iHTaskID.ToString());
				this.ClearForceCompleteChain();
				return true;
			}
			Logger.Log("ForceCompleteV2: retry task " + me.m_iHTaskID.ToString() + " err " + forceErrorCode.ToString() + " attempt " + this.m_iForceCompleteRetryCount.ToString());
			this.m_iForceCompletePendingTaskId = me.m_iHTaskID;
			this.PrepareTaskForForceComplete(task);
			if (0 < me.m_iSTGrantTimer || 0 < me.m_iCSUCheckTimer)
			{
				task.SetRemainTime(0f);
				task.m_fLifeTime = 0f;
				if (this.IsExistTaskInActiveMission(me.m_iHTaskID) && task.GetMissionState() == 1)
				{
					this.m_iForceCompletePendingTaskId = 0;
					this.RequestForceCompleteTaskEnd(task);
					return true;
				}
				this.m_iForceCompletePendingTaskId = me.m_iHTaskID;
				return true;
			}
			if (!this.IsExistTaskInActiveMission(me.m_iHTaskID) || task.GetMissionState() != 1)
			{
				this.RequestTaskStart(me.m_iHTaskID, 0);
			}
			else
			{
				this.RequestForceCompleteTaskEnd(task);
			}
			return true;
		}
		Logger.Log("ForceCompleteV2: abort task " + me.m_iHTaskID.ToString() + " err " + forceErrorCode.ToString());
		this.ClearForceCompleteChain();
		return true;
	}

	private bool ForceCompleteOnEndSucc(cnMissionNode task)
	{
		if (!this.bForceCompleteChain || task == null || task.GetMe() == null)
		{
			return false;
		}
		cnMissionNode nextChainTask = null;
		if (0 < task.GetMe().m_iSUOutgoingTask)
		{
			nextChainTask = this.GetTask(task.GetMe().m_iSUOutgoingTask);
		}
		if (nextChainTask == null)
		{
			nextChainTask = this.GetNextTaskNode(task);
		}
		if (nextChainTask != null && nextChainTask.GetMe() != null)
		{
			Logger.Log("ForceCompleteV2: advance to task " + nextChainTask.GetMe().m_iHTaskID.ToString());
			this.TryForceCompleteChainTask(nextChainTask);
			return true;
		}
		Logger.Log("ForceCompleteV2: chain finished");
		this.ClearForceCompleteChain();
		return true;
	}

	private void ForceCompleteOnStartFail(int iTaskNum)
	{
		if (!this.bForceCompleteChain)
		{
			return;
		}
		cnMissionNode task = this.GetTask(iTaskNum);
		if (task == null || task.GetMe() == null)
		{
			return;
		}
		MissionElement me = task.GetMe();
		if (0 < me.m_iRequireInstanceID && !this.IsExistTaskInActiveMission(me.m_iHTaskID))
		{
			Logger.Log("ForceCompleteV2: instance start fail complete task " + iTaskNum.ToString());
			this.PrepareTaskForForceComplete(task);
			this.m_iForceCompletePendingTaskId = 0;
			this.ForceCompleteEmitTaskEndPacket(iTaskNum, 0, true);
			return;
		}
		if (this.m_iForceCompletePendingTaskId != 0 && this.m_iForceCompletePendingTaskId != iTaskNum)
		{
			this.m_iForceCompletePendingTaskId = 0;
		}
		if (this.m_iForceCompletePendingTaskId == 0 || this.m_iForceCompletePendingTaskId != iTaskNum)
		{
			return;
		}
		if (this.IsExistTaskInActiveMission(me.m_iHTaskID) && task.GetMissionState() == 1)
		{
			this.PrepareTaskForForceComplete(task);
			if (0 < me.m_iSTGrantTimer || 0 < me.m_iCSUCheckTimer)
			{
				task.SetRemainTime(0f);
				task.m_fLifeTime = 0f;
				this.m_iForceCompletePendingTaskId = me.m_iHTaskID;
				return;
			}
			this.m_iForceCompletePendingTaskId = 0;
			Logger.Log("ForceCompleteV2: complete after start fail " + iTaskNum.ToString());
			bool sendError = 0 < me.m_iRequireInstanceID;
			this.ForceCompleteEmitTaskEndPacket(iTaskNum, 0, sendError);
		}
		else
		{
			this.ClearForceCompleteChain();
		}
	}

	private void ForceCompleteOnStartSucc(int iTaskNum)
	{
		if (!this.bForceCompleteChain)
		{
			return;
		}
		if (this.m_iForceCompletePendingTaskId != 0 && this.m_iForceCompletePendingTaskId != iTaskNum)
		{
			this.m_iForceCompletePendingTaskId = 0;
		}
		cnMissionNode task = this.GetTask(iTaskNum);
		if (task == null || task.GetMe() == null)
		{
			return;
		}
		this.PrepareTaskForForceComplete(task);
		MissionElement forceMe = task.GetMe();
		if (0 < forceMe.m_iSTGrantTimer || 0 < forceMe.m_iCSUCheckTimer)
		{
			task.SetRemainTime(0f);
			task.m_fLifeTime = 0f;
			this.m_iForceCompletePendingTaskId = iTaskNum;
			Logger.Log("ForceCompleteV2: wait timer after start " + iTaskNum.ToString());
			return;
		}
		this.m_iForceCompletePendingTaskId = 0;
		bool sendError = 0 < forceMe.m_iRequireInstanceID;
		Logger.Log("ForceCompleteV2: complete after start " + iTaskNum.ToString() + " err " + sendError.ToString());
		this.ForceCompleteEmitTaskEndPacket(iTaskNum, 0, sendError);
	}
